Testing case insensitive: set both to all uppercase or all lowercase

function areEqualCaseInsensitive(str1, str2) {
return str1.toUpperCase() === str2.toUpperCase();
}

console.log(areEqualCaseInsensitive("hi", "Hi"));

Test if a string starts or ends with a particular substring

const beverageType = "whiskey sour";

if (beverageType.endsWith('sour')) {
console.log("It's a sour.");
} else {
console.log("It's not a sour.")
}

"Whiskey sour" will return "It's a sour."
"Lemonade" will return "It's not a sour."

Find the position of a substring

const duckBoys = "Huey, Dewey, and Louie";

console.log("'Dewey' starts at index " + duckBoys.indexOf("Dewey"));

console.log("'Donald' is not in the string, so 'Donald' returns " + duckBoys.indexOf("Donald"));
